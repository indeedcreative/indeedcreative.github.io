Connect Nginx server to Flask server

Example project folder

/var/www/lab_app

Make a sample Flask server app

#hello.py
from flask import Flask
app = Flask(__name__)

@app.route("/")
def hello():
	return "Hello World"

@app.route("/example")
def example_route():
	return "This is an example route"

if __name__ == "__main__":
	app.run(host="0.0.0.0",port=8080)

Start from root 

su

Install Uwsgi

#within virtual_env
pip install uwsgi

-Nginx Configuration

Remove the default page configuration 

rm /etc/nginx/sites-enabled/default

Make new configuration file

nano /var/www/lab_app/lab_app_nginx.conf
#Full path: /var/www/lab_app/lab_app_nginx.conf

server {
    listen      80;
    server_name localhost;
    charset     utf-8;
    client_max_body_size 75M;

    location /static {
        root /var/www/lab_app/;
    }

    location / { try_files $uri @labapp; }
    location @labapp {
        include uwsgi_params;
        uwsgi_pass unix:/var/www/lab_app/lab_app_uwsgi.sock; 
				#socket file to connect nginx server via uwsgi socket
    }
}

Link the file with files inside /etc/nginx/sites-enabled directory

ln -s /var/www/lab_app/lab_app_nginx.conf /etc/nginx/conf.d/
ls -al /etc/nginx/conf.d #check the file linked together
/etc/init.d/nginx restart #load the configuration file
#check message [ok] Restarting nginx(via systemctl): nginx.service

-USWGI Configuration

Link the file with files inside /etc/nginx/sites-enabled directory

nano lab_app_uwsgi.ini
#Full path: /var/www/lab_app/lab_app_uwsgi.ini

[uwsgi]
#application's base folder
base = /var/www/lab_app

#python module to import
app = hello
module = %(app)

home = %(base)
pythonpath = %(base)

#socket file's location
#%n : file name without extension
socket = /var/www/lab_app/%n.sock

#permissions for the socket file
chmod-socket    = 666

#the variable that holds a flask application inside the module imported at line #6
callable = app

#location of log files
logto = /var/log/uwsgi/%n.log

Make log folder

#location of log files
mkdir /var/log/uwsgi

-UWSGI and NGINX configuration Testing

bin/uwsgi --ini /var/www/lab_app/lab_app_uwsgi.ini

Make the server start automatically (systemd)

nano /etc/systemd/system/emperor.uwsgi.service
[Unit]
Description=uWSGI Emperor
After=syslog.target

[Service]
ExecStart=/var/www/lab_app/bin/uwsgi --ini /var/www/lab_app/lab_app_uwsgi.ini
# Requires systemd version 211 or newer
RuntimeDirectory=uwsgi
Restart=always
KillSignal=SIGQUIT
Type=notify
StandardError=syslog
NotifyAccess=all

[Install]
WantedBy=multi-user.target
systemctl start emperor.uwsgi.service
systemctl enable emperor.uwsgi.service #automatically start server when rebooted
systemctl status emperor.uwsgi.service
root@raspberrypi:/var/www/lab_app# systemctl restart emperor.uwsgi.service
//restart the service after making a change
